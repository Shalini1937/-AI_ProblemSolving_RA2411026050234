/* ════════════════════════════════════════════
   CSP ENGINE
   ════════════════════════════════════════════ */
const COLORS = ['Red','Green','Blue','Yellow'];
const COLOR_HEX = { Red:'#ff4757', Green:'#2ed573', Blue:'#1e90ff', Yellow:'#ffa502' };

function isValid(region, color, assignment, neighbors) {
  for (const nb of (neighbors[region] || [])) {
    if (assignment[nb] === color) return false;
  }
  return true;
}

function backtrack(regions, neighbors, nc, assignment, stats) {
  if (Object.keys(assignment).length === regions.length) return true;
  const region = regions.find(r => !(r in assignment));
  for (const color of COLORS.slice(0, nc)) {
    stats.checks++;
    if (isValid(region, color, assignment, neighbors)) {
      assignment[region] = color;
      stats.steps.push({ region, color, action: 'assign' });
      if (backtrack(regions, neighbors, nc, assignment, stats)) return true;
      delete assignment[region];
      stats.steps.push({ region, color, action: 'backtrack' });
    }
  }
  return false;
}

function solve(regions, neighbors, nc) {
  const stats = { checks: 0, steps: [] };
  const assignment = {};
  const t0 = performance.now();
  const ok = backtrack(regions, neighbors, nc, assignment, stats);
  return { ok, assignment: ok ? assignment : {}, checks: stats.checks, steps: stats.steps, ms: (performance.now()-t0).toFixed(2) };
}

function parseRegions(raw) {
  return raw.split(',').map(s => s.trim().toUpperCase()).filter(Boolean);
}

function parseNeighbors(raw, regions) {
  const nb = {};
  regions.forEach(r => nb[r] = []);
  for (const line of raw.split('\n')) {
    const t = line.trim(); if (!t) continue;
    let parts;
    if (t.includes('→')) parts = t.split('→');
    else if (t.includes('->')) parts = t.split('->');
    else if (t.includes(':')) parts = t.split(':');
    else continue;
    if (parts.length !== 2) continue;
    const src = parts[0].trim().toUpperCase();
    const tgts = parts[1].split(',').map(s => s.trim().toUpperCase()).filter(Boolean);
    if (!nb[src]) nb[src] = [];
    for (const tgt of tgts) {
      if (!nb[tgt]) nb[tgt] = [];
      if (!nb[src].includes(tgt)) nb[src].push(tgt);
      if (!nb[tgt].includes(src)) nb[tgt].push(src);
    }
  }
  return nb;
}

const svg = d3.select('#gSvg');
let sim, gNodes = [], gLinks = [], lSel, nSel;

function buildEdges(neighborMap) {
  const edges = [], seen = new Set();
  Object.entries(neighborMap).forEach(([src, targets]) => {
    targets.forEach(tgt => {
      const key = [src,tgt].sort().join('|');
      if (!seen.has(key)) { seen.add(key); edges.push([src, tgt]); }
    });
  });
  return edges;
}

function renderGraph(nodeIds, edges, assignment) {
  svg.selectAll('*').remove();
  const box = document.getElementById('graphArea');
  const W = box.clientWidth, H = box.clientHeight;
  const r0 = Math.min(W, H) * .34;

  gNodes = nodeIds.map((id, i) => {
    const a = 2 * Math.PI * i / nodeIds.length - Math.PI / 2;
    return { id, color: assignment[id] || null, x: W/2 + r0*Math.cos(a), y: H/2 + r0*Math.sin(a) };
  });

  const seenE = new Set();
  gLinks = [];
  edges.forEach(([a,b]) => {
    const k = [a,b].sort().join('|');
    if (!seenE.has(k)) { seenE.add(k); gLinks.push({ source: a, target: b }); }
  });

  lSel = svg.append('g').selectAll('line').data(gLinks).join('line').attr('class','lnk');

  nSel = svg.append('g').selectAll('g').data(gNodes).join('g').attr('class','nd')
    .call(d3.drag()
      .filter(ev => !ev.shiftKey)
      .on('start', (ev,d) => { if(!ev.active) sim.alphaTarget(.3).restart(); d.fx=d.x; d.fy=d.y; })
      .on('drag',  (ev,d) => { d.fx=ev.x; d.fy=ev.y; })
      .on('end',   (ev,d) => { if(!ev.active) sim.alphaTarget(0); d.fx=null; d.fy=null; })
    );

  nSel.append('circle')
    .attr('r', 22)
    .attr('fill', d => d.color ? COLOR_HEX[d.color] : 'rgba(255,255,255,.05)')
    .attr('stroke', d => d.color ? COLOR_HEX[d.color] : 'rgba(255,255,255,.15)')
    .attr('fill-opacity', d => d.color ? .85 : 1);

  nSel.append('text').text(d => d.id)
    .attr('fill', d => d.color && ['Yellow'].includes(d.color) ? '#111' : 'white');

  if (sim) sim.stop();
  sim = d3.forceSimulation(gNodes)
    .force('link', d3.forceLink(gLinks).id(d=>d.id).distance(100))
    .force('charge', d3.forceManyBody().strength(-260))
    .force('center', d3.forceCenter(W/2, H/2))
    .force('collision', d3.forceCollide(36))
    .on('tick', () => {
      lSel.attr('x1',d=>d.source.x).attr('y1',d=>d.source.y)
          .attr('x2',d=>d.target.x).attr('y2',d=>d.target.y);
      nSel.attr('transform', d=>`translate(${d.x},${d.y})`);
    });

  document.getElementById('graphEmpty').style.display = 'none';
}

function animateColors(steps) {
  const assigns = steps.filter(s => s.action === 'assign');
  assigns.forEach((s, i) => {
    setTimeout(() => {
      nSel.select('circle').filter(d => d.id === s.region)
        .transition().duration(350)
        .attr('fill', COLOR_HEX[s.color])
        .attr('stroke', COLOR_HEX[s.color])
        .attr('fill-opacity', .85);
      nSel.select('text').filter(d => d.id === s.region)
        .attr('fill', s.color === 'Yellow' ? '#111' : 'white');
      const nd = gNodes.find(n => n.id === s.region);
      if (nd) nd.color = s.color;
    }, i * 220);
  });
}

  const id = label.trim().toUpperCase();
  if (gNodes.find(n => n.id === id)) { showToast('⚠️ Region already exists'); return; }
  gNodes.push({ id, x: mx, y: my, color: null });
  syncForm(); rebuildCanvas();
  showToast('✅ Node added: ' + id);
});
svg.on('mousedown.s', function(ev) {
  if (!ev.shiftKey) return;
  const [mx,my] = d3.pointer(ev, this);
  shiftSrc = gNodes.find(n => Math.hypot(n.x-mx, n.y-my) < 30) || null;
});
svg.on('mouseup.s', function(ev) {
  if (!shiftSrc) return;
  const [mx,my] = d3.pointer(ev, this);
  const tgt = gNodes.find(n => n.id !== shiftSrc.id && Math.hypot(n.x-mx, n.y-my) < 30);
  if (tgt) {
    const ex = gLinks.some(l => {
      const s = l.source.id||l.source, t = l.target.id||l.target;
      return (s===shiftSrc.id&&t===tgt.id)||(s===tgt.id&&t===shiftSrc.id);
    });
    if (!ex) {
      gLinks.push({ source: shiftSrc.id, target: tgt.id });
      syncForm(); rebuildCanvas();
      showToast(`🔗 Edge: ${shiftSrc.id} ↔ ${tgt.id}`);
    }
  }
  shiftSrc = null;
});

function syncForm() {
  document.getElementById('regions').value = gNodes.map(n=>n.id).join(', ');
  const seen = new Set(), lines = [];
  gLinks.forEach(l => {
    const s=l.source.id||l.source, t=l.target.id||l.target;
    const k=[s,t].sort().join('|');
    if (!seen.has(k)) { seen.add(k); lines.push(`${s} → ${t}`); }
  });
  document.getElementById('neighbors').value = lines.join('\n');
}

function rebuildCanvas() {
  if (!gNodes.length) return;
  const a = {};
  gNodes.forEach(n => { if(n.color) a[n.id]=n.color; });
  renderGraph(gNodes.map(n=>n.id), gLinks.map(l=>[l.source.id||l.source, l.target.id||l.target]), a);
}

  gNodes=[]; gLinks=[];
  document.getElementById('graphEmpty').style.display='flex';
  document.getElementById('resultArea').innerHTML='';
  showToast('🗑️ Graph cleared');
};

   SOLVE

let lastResult = null;

document.getElementById('solveBtn').onclick = function() {
  const regRaw = document.getElementById('regions').value.trim();
  const nbRaw  = document.getElementById('neighbors').value.trim();
  const nc     = +document.getElementById('numColors').value;

  if (!regRaw) { showToast('⚠️ Enter at least one region'); return; }

  document.getElementById('solveLbl').textContent = 'Solving…';
  document.getElementById('solveSpinner').hidden = false;
  this.disabled = true;
  document.getElementById('resultArea').innerHTML = '';

  setTimeout(() => {
    const regions   = parseRegions(regRaw);
    const neighbors = parseNeighbors(nbRaw, regions);
    const result    = solve(regions, neighbors, nc);
    lastResult      = { result, regions, neighbors, nc };

    const edges = buildEdges(neighbors);
    renderGraph(regions, edges, {});

    if (result.ok) {
      setTimeout(() => animateColors(result.steps), 300);
      renderResult(result);
    } else {
      renderError(nc);
    }

    document.getElementById('solveLbl').textContent = 'Solve with AI Backtracking';
    document.getElementById('solveSpinner').hidden = true;
    this.disabled = false;

    // scroll to result
    setTimeout(() => {
      document.getElementById('resultArea').scrollIntoView({ behavior: 'smooth', block: 'nearest' });
    }, 400);
  }, 80);
};
